pipeline {
    agent any
    triggers {
        GenericTrigger(
            genericVariables: [[key: 'repo_url', value: '$.repo_url']],
            token: 'REPO_CLONE_TOKEN'
        )
    }
    stages {
        stage('Clone') {
            steps {
                sh "git clone ${env.repo_url} && ls -la"
            }
        }
    }
}
